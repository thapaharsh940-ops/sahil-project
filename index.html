from flask import Flask, request, jsonify, send_from_directory, render_template, redirect, url_for, session from werkzeug.security import generate_password_hash, check_password_hash import os, json from functools import wraps app = Flask(__name__) app.secret_key = 'change_this_secret_key' # Change in production! DATA_DIR = 'data' UPLOAD_DIR = 'uploads' USERS_FILE = os.path.join(DATA_DIR, 'users.json') os.makedirs(DATA_DIR, exist_ok=True) os.makedirs(UPLOAD_DIR, exist_ok=True) # initialize users file if not os.path.exists(USERS_FILE): with open(USERS_FILE, 'w') as f: json.dump({'users': []}, f) def load_users(): with open(USERS_FILE) as f: return json.load(f) def save_users(data): with open(USERS_FILE, 'w') as f: json.dump(data, f, indent=2) def login_required(f): @wraps(f) def decorated(*args, **kwargs): if 'username' not in session: return redirect(url_for('login_page')) return f(*args, **kwargs) return decorated @app.route('/') def index(): return redirect(url_for('login_page')) @app.route('/login', methods=['GET', 'POST']) def login_page(): if request.method == 'POST': username = request.form.get('username') password = request.form.get('password') db = load_users() user = next((u for u in db['users'] if u['username'] == username), None) if user and check_password_hash(user['password_hash'], password): session['username'] = username return redirect(url_for('downloads_page')) else: return render_template('login.html', error="Invalid username or password") return render_template('login.html') @app.route('/register', methods=['POST']) def register(): username = request.form.get('username') password = request.form.get('password') db = load_users() if any(u['username'] == username for u in db['users']): return render_template('login.html', error="Username already exists") db['users'].append({ 'username': username, 'password_hash': generate_password_hash(password) }) save_users(db) return render_template('login.html', message="Registered successfully! Now log in.") @app.route('/downloads') @login_required def downloads_page(): files = os.listdir(UPLOAD_DIR) return render_template('downloads.html', files=files, username=session['username']) @app.route('/upload', methods=['POST']) @login_required def upload_file(): file = request.files.get('file') if not file: return redirect(url_for('downloads_page')) file.save(os.path.join(UPLOAD_DIR, file.filename)) return redirect(url_for('downloads_page')) @app.route('/download/') @login_required def download(filename): return send_from_directory(UPLOAD_DIR, filename, as_attachment=True) @app.route('/logout') def logout(): session.clear() return redirect(url_for('login_page')) if __name__ == '__main__': app.run(debug=True)
